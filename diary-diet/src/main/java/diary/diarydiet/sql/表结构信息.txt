CREATE TABLE `diet_records` (
  `id`              BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  `user_id`         BIGINT UNSIGNED NOT NULL COMMENT '用户ID（关联用户中心）',
  `eat_time`        DATETIME        NOT NULL COMMENT '食用时间',
  `meal_type`       TINYINT         NOT NULL COMMENT '餐别：10早餐 15早加餐 20午餐 25午加餐 30晚餐 35夜宵',
  `food_name`       VARCHAR(200)    NOT NULL COMMENT '食物/菜品名称',
  `calories`        INT UNSIGNED    NOT NULL DEFAULT 0 COMMENT '热量（千卡）',
  `protein`         DECIMAL(8,2)    NOT NULL DEFAULT 0.00 COMMENT '蛋白质(g)',
  `fat`             DECIMAL(8,2)    NOT NULL DEFAULT 0.00 COMMENT '脂肪(g)',
  `carbohydrate`    DECIMAL(8,2)    NOT NULL DEFAULT 0.00 COMMENT '碳水化合物(g)',
  `fullness_score`  TINYINT         NULL COMMENT '饱腹感评分 1~10',
  `location`        VARCHAR(100)    NULL COMMENT '用餐地点（家里/公司/餐厅等）',
  `note`            VARCHAR(500)    NULL COMMENT '备注',
  `deleted`         TINYINT(1)      NOT NULL DEFAULT 0 COMMENT '逻辑删除 0正常 1删除',
  `created_at`      DATETIME(3)     NOT NULL DEFAULT CURRENT_TIMESTAMP(3),
  `updated_at`      DATETIME(3)     NOT NULL DEFAULT CURRENT_TIMESTAMP(3) ON UPDATE CURRENT_TIMESTAMP(3),
  PRIMARY KEY (`id`),
  KEY `idx_user_eat_time` (`user_id`, `eat_time`),
  KEY `idx_user_meal` (`user_id`, `meal_type`, `eat_time`),
  KEY `idx_deleted` (`deleted`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='饮食记录表';
